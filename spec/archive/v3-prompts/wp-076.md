
Avatar design:
- 28px circle, `border-radius: 50%`
- Background: `participant.color` (already a `var(--collab-color-N)`)
- Initials: `var(--text-xs)`, `var(--color-on-accent)` or white
- Online: subtle green dot (2px) at bottom-right of circle — use `var(--color-success)` token
- Offline: reduce opacity to 0.5
- Local user (self): ring border in `var(--color-accent)`, 2px
- Overflow chip: `var(--color-surface-2)` background, `var(--text-xs)`

### 4. Zustand presence store

Add a `presenceStore.ts` in `packages/web/src/`:
```typescript
import { create } from 'zustand';
import type { Participant } from '@bim-ai/core';

type PresenceStore = {
  participants: Participant[];
  localUserId: string | null;
  setParticipants: (participants: Participant[]) => void;
  setLocalUserId: (userId: string) => void;
  addParticipant: (p: Participant) => void;
  removeParticipant: (userId: string) => void;
  updateParticipant: (userId: string, patch: Partial<Participant>) => void;
};

export const usePresenceStore = create<PresenceStore>((set) => ({
  participants: [],
  localUserId: null,
  setParticipants: (participants) => set({ participants }),
  setLocalUserId: (localUserId) => set({ localUserId }),
  addParticipant: (p) => set((s) => ({ participants: [...s.participants.filter(x => x.userId !== p.userId), p] })),
  removeParticipant: (userId) => set((s) => ({ participants: s.participants.filter(p => p.userId !== userId) })),
  updateParticipant: (userId, patch) => set((s) => ({
    participants: s.participants.map(p => p.userId === userId ? { ...p, ...patch } : p),
  })),
}));
```
