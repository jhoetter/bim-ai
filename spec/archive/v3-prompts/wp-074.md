Actual external-service integrations are vision. Per §L, no rendering model lives in the binary.

```python
"""EXP-V3-01 — deterministic render-pipeline export tool.

Exports: gltf, gltf-pbr, ifc-bundle, metadata-only.
Each format produces a RenderExportBundle JSON + the primary asset file.
"""
from __future__ import annotations
from dataclasses import dataclass, field, asdict
from typing import Literal, Optional
import json
import time

ExportFormat = Literal["gltf", "gltf-pbr", "ifc-bundle", "metadata-only"]

@dataclass
class CameraEntry:
    view_id: str
    position_mm: dict  # {xMm, yMm, zMm}
    target_mm: dict    # {xMm, yMm, zMm}
    fov_deg: float = 60.0

@dataclass
class SunSettings:
    azimuth_deg: float = 180.0
    elevation_deg: float = 45.0
    intensity: float = 1.0

@dataclass
class MaterialEntry:
    id: str
    pbr: dict  # {baseColorHex, metallicFactor, roughnessFactor, ...}

@dataclass
class AnnotationEntry:
    id: str
    text: str
    position_mm: dict  # {xMm, yMm, zMm}

@dataclass
class RenderExportBundle:
    schema_version: str = "exp-v3.0"
    format: ExportFormat = "metadata-only"
    primary_asset: Optional[dict] = None  # {kind, pathInArchive}
    metadata: dict = field(default_factory=lambda: {
        "cameras": [],
        "sunSettings": asdict(SunSettings()),
        "materials": [],
        "annotations": [],
    })
    export_timestamp: str = field(default_factory=lambda: time.strftime("%Y-%m-%dT%H:%M:%SZ"))

    def to_dict(self) -> dict:
        return {
            "schemaVersion": self.schema_version,
            "format": self.format,
            "primaryAsset": self.primary_asset,
            "metadata": self.metadata,
            "exportTimestamp": self.export_timestamp,
        }


def build_export_bundle(
    model_state: dict,
    fmt: ExportFormat,
    view_id: Optional[str] = None,
) -> RenderExportBundle:
    """Build a RenderExportBundle from model state."""
    elements = model_state.get("elements", [])
    
    # Collect cameras from viewpoint elements or saved views
    cameras = []
    for el in elements:
        if el.get("kind") in ("viewpoint", "saved_view"):
            cameras.append({
                "viewId": el.get("id", ""),
                "positionMm": el.get("cameraState", {}).get("positionMm", {"xMm": 0, "yMm": -10000, "zMm": 3000}),
                "targetMm": el.get("cameraState", {}).get("targetMm", {"xMm": 0, "yMm": 0, "zMm": 0}),
                "fovDeg": el.get("cameraState", {}).get("fovDeg", 60.0),
            })
    
    # Filter to specific view if requested
    if view_id:
        cameras = [c for c in cameras if c["viewId"] == view_id]
    
    # Collect materials
    materials = []
    for el in elements:
        if el.get("kind") == "material":
            materials.append({
                "id": el.get("id", ""),
                "pbr": el.get("pbr", {}),
            })
    
    # Sun settings — from project settings if present
    sun = {"azimuthDeg": 180.0, "elevationDeg": 45.0, "intensity": 1.0}
    for el in elements:
        if el.get("kind") == "project_settings":
            if "sunAzimuthDeg" in el:
                sun["azimuthDeg"] = el["sunAzimuthDeg"]
            if "sunElevationDeg" in el:
                sun["elevationDeg"] = el["sunElevationDeg"]
    
    primary_asset = None
    if fmt in ("gltf", "gltf-pbr"):
        primary_asset = {"kind": "gltf", "pathInArchive": "model.glb"}
    elif fmt == "ifc-bundle":
        primary_asset = {"kind": "ifc", "pathInArchive": "model.ifc"}
    
    bundle = RenderExportBundle(format=fmt, primary_asset=primary_asset)
    bundle.metadata = {
        "cameras": cameras,
        "sunSettings": sun,
        "materials": materials,
        "annotations": [],
    }
    return bundle
```
