  ## Gable height at a point (x, z) both in metres
    if (ridgeAlongX):
      halfSpan = (oz1 - oz0) / 2
      ridgeY = eaveY + halfSpan * tan(slopeRad)
      height = eaveY + max(0, halfSpan - |z - (oz0+oz1)/2|) * tan(slopeRad)
    else:
      halfSpan = (ox1 - ox0) / 2
      ridgeY = eaveY + halfSpan * tan(slopeRad)
      height = eaveY + max(0, halfSpan - |x - (ox0+ox1)/2|) * tan(slopeRad)

  ## Hip height at a point (x, z) both in metres
    if (ridgeAlongX):
      halfSpanZ = (oz1 - oz0) / 2
      ridgeY = eaveY + halfSpanZ * tan(slopeRad)
      dz = halfSpanZ - |z - (oz0+oz1)/2|
      rx0 = ox0 + halfSpanZ;  rx1 = ox1 - halfSpanZ
      if (rx0 >= rx1):   // pyramid — use min of both spans
        dx = (ox1 - ox0) / 2 - |x - (ox0+ox1)/2|
        height = eaveY + max(0, min(dz, dx)) * tan(slopeRad)
      else:              // full hip
        dx = (x < rx0) ? (x - ox0) : (x > rx1) ? (ox1 - x) : halfSpanZ
        height = eaveY + max(0, min(dz, dx)) * tan(slopeRad)
    else: (symmetric — swap x/z roles)
